# Demo Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `jmux --demo` that launches an isolated, ephemeral jmux instance with mock Linear/GitLab adapters and pre-seeded data showcasing all features.

**Architecture:** Mock adapters implement the existing `CodeHostAdapter` and `IssueTrackerAdapter` interfaces, backed by in-memory maps from a static seed data module. A setup function creates a temp directory with minimal git repos, spawns tmux sessions on an isolated socket, and returns the adapters + paths. `main.ts` threads these into the existing startup flow. Cleanup kills the demo tmux server and removes the temp dir.

**Tech Stack:** TypeScript, Bun APIs (`Bun.spawnSync`, `rmSync`), existing adapter interfaces from `src/adapters/types.ts`

---

import type {
  CodeHostAdapter,
  AdapterAuthState,
  MergeRequest,
  BranchContext,
} from "../adapters/types";
import { DEMO_MRS, DEMO_REVIEW_MR_IDS } from "./seed-data";

export class DemoCodeHostAdapter implements CodeHostAdapter {
  type = "demo";
  authState: AdapterAuthState = "ok";
  authHint = "demo mode — no credentials needed";

  private mrs: Map<string, MergeRequest>;
  private byBranch: Map<string, MergeRequest>;

  constructor() {
    this.mrs = new Map(DEMO_MRS.map(mr => [mr.id, { ...mr }]));
    this.byBranch = new Map(DEMO_MRS.map(mr => [mr.sourceBranch, { ...mr }]));
  }

  async authenticate(): Promise<void> {}

  async getMergeRequest(remote: string, branch: string): Promise<MergeRequest | null> {
    return this.byBranch.get(branch) ?? null;
  }

  async pollMergeRequest(mrId: string): Promise<MergeRequest> {
    const mr = this.mrs.get(mrId);
    if (!mr) throw new Error(`MR not found: ${mrId}`);
    return { ...mr };
  }

  async pollAllMergeRequests(remotes: BranchContext[]): Promise<Map<string, MergeRequest>> {
    const result = new Map<string, MergeRequest>();
    for (const ctx of remotes) {
      const mr = this.byBranch.get(ctx.branch);
      if (mr) result.set(ctx.sessionName, { ...mr });
    }
    return result;
  }

  async pollMergeRequestsByIds(ids: string[]): Promise<Map<string, MergeRequest>> {
    const result = new Map<string, MergeRequest>();
    for (const id of ids) {
      const mr = this.mrs.get(id);
      if (mr) result.set(id, { ...mr });
    }
    return result;
  }

  openInBrowser(_mrId: string): void {}

  async markReady(mrId: string): Promise<void> {
    const mr = this.mrs.get(mrId);
    if (mr && mr.status === "draft") {
      mr.status = "open";
      this.byBranch.set(mr.sourceBranch, mr);
    }
  }

  async approve(mrId: string): Promise<void> {
    const mr = this.mrs.get(mrId);
    if (mr) {
      mr.approvals = { ...mr.approvals, current: mr.approvals.current + 1 };
      this.byBranch.set(mr.sourceBranch, mr);
    }
  }

  async searchMergeRequests(query: string): Promise<MergeRequest[]> {
    const q = query.toLowerCase();
    return [...this.mrs.values()].filter(
      mr => mr.title.toLowerCase().includes(q) || mr.sourceBranch.toLowerCase().includes(q)
    );
  }

  parseMrUrl(url: string): string | null {
    const match = url.match(/gitlab\.com\/(.+?)\/-\/merge_requests\/(\d+)/);
    if (!match) return null;
    return `${encodeURIComponent(match[1])}:${match[2]}`;
  }

  async getMyMergeRequests(): Promise<MergeRequest[]> {
    return [...this.mrs.values()];
  }

  async getMrsAwaitingMyReview(): Promise<MergeRequest[]> {
    return [...this.mrs.values()].filter(mr => DEMO_REVIEW_MR_IDS.has(mr.id));
  }
}

import type {
  IssueTrackerAdapter,
  AdapterAuthState,
  Issue,

import { DEMO_ISSUES, DEMO_TEAMS } from "./seed-data";

export class DemoIssueTrackerAdapter implements IssueTrackerAdapter {
  type = "demo";
  authState: AdapterAuthState = "ok";
  authHint = "demo mode — no credentials needed";

  private issues: Map<string, Issue>;
  private byBranch: Map<string, Issue>;

  constructor() {
    this.issues = new Map(DEMO_ISSUES.map(i => [i.id, { ...i }]));
    this.byBranch = new Map(
      DEMO_ISSUES.filter(i => i.branchName).map(i => [i.branchName!, { ...i }])
    );
  }

  async authenticate(): Promise<void> {}

  async getLinkedIssue(mrUrl: string): Promise<Issue | null> {
    for (const issue of this.issues.values()) {
      if (issue.linkedMrUrls.includes(mrUrl)) return { ...issue };
    }
    return null;
  }

  async getIssueByBranch(branch: string): Promise<Issue | null> {
    return this.byBranch.get(branch) ? { ...this.byBranch.get(branch)! } : null;
  }

  async pollIssue(issueId: string): Promise<Issue> {
    const issue = this.issues.get(issueId);
    if (!issue) throw new Error(`Issue not found: ${issueId}`);
    return { ...issue };
  }

  async pollAllIssues(issueIds: string[]): Promise<Map<string, Issue>> {
    const result = new Map<string, Issue>();
    for (const id of issueIds) {
      const issue = this.issues.get(id);
      if (issue) result.set(id, { ...issue });
    }
    return result;
  }

  async getAvailableStatuses(_issueId: string): Promise<string[]> {
    return ["Backlog", "Todo", "In Progress", "In Review", "Done"];
  }

  openInBrowser(_issueId: string): void {}

  async updateStatus(issueId: string, status: string): Promise<void> {
    const issue = this.issues.get(issueId);
    if (issue) {
      issue.status = status;
      if (issue.branchName) this.byBranch.set(issue.branchName, issue);
    }
  }

  async searchIssues(query: string): Promise<Issue[]> {
    const q = query.toLowerCase();
    return [...this.issues.values()].filter(
      i => i.title.toLowerCase().includes(q) || i.identifier.toLowerCase().includes(q)
    );
  }

  async getMyIssues(): Promise<Issue[]> {
    return [...this.issues.values()];
  }

  async getTeams(): Promise<Array<{ id: string; name: string }>> {
    return DEMO_TEAMS.map(t => ({ ...t }));
  }
}

function initGitRepo(dir: string, branch: string, remote: string): void {
  const opts = { cwd: dir, stdout: "pipe" as const, stderr: "pipe" as const };
  Bun.spawnSync(["git", "init"], opts);
  Bun.spawnSync(["git", "checkout", "-b", branch], opts);
  Bun.spawnSync(["git", "remote", "add", "origin", remote], opts);
  Bun.spawnSync(
    ["git", "-c", "user.name=Demo", "-c", "user.email=demo@jmux.dev", "commit", "--allow-empty", "-m", "init"],
    opts,
  );
}

export function setupDemo(): DemoContext {
  const pid = process.pid;
  const socketName = `jmux-demo-${pid}`;
  const tmpDir = `/tmp/jmux-demo-${pid}`;

  // 1. Create temp directory structure
  mkdirSync(tmpDir, { recursive: true });
  for (const session of DEMO_SESSIONS) {
    const sessionDir = resolve(tmpDir, "sessions", session.group, session.name);
    mkdirSync(sessionDir, { recursive: true });
    initGitRepo(sessionDir, session.branch, session.remote);
  }

  // 2. Write demo config
  const configPath = resolve(tmpDir, "config.json");
  const config = {
    sidebarWidth: 26,
    cacheTimers: false,
    adapters: {
      codeHost: { type: "demo" },
      issueTracker: { type: "demo" },
    },
  };
  writeFileSync(configPath, JSON.stringify(config, null, 2) + "\n");

  // 3. Write demo state with manual links
  const statePath = resolve(tmpDir, "state.json");
  const sessionLinks: Record<string, Array<{ type: string; id: string }>> = {};
  for (const [sessionName, links] of Object.entries(DEMO_MANUAL_LINKS)) {
    sessionLinks[sessionName] = links;
  }
  writeFileSync(statePath, JSON.stringify({ sessionLinks }, null, 2) + "\n");

  // 4. Create tmux sessions on the demo socket
  const tmuxOpts = { stdout: "pipe" as const, stderr: "pipe" as const };
  for (let i = 0; i < DEMO_SESSIONS.length; i++) {
    const session = DEMO_SESSIONS[i];
    const sessionDir = resolve(tmpDir, "sessions", session.group, session.name);
    if (i === 0) {
      // First session — create the server
      Bun.spawnSync(
        ["tmux", "-L", socketName, "new-session", "-d", "-s", session.name, "-c", sessionDir],
        tmuxOpts,
      );
    } else {
      Bun.spawnSync(
        ["tmux", "-L", socketName, "new-session", "-d", "-s", session.name, "-c", sessionDir],
        tmuxOpts,
      );
    }

    // Set attention flag if needed
    if (session.attention) {
      Bun.spawnSync(
        ["tmux", "-L", socketName, "set-option", "-t", session.name, "@jmux-attention", "1"],
        tmuxOpts,
      );
    }
  }

  // 5. Create adapters
  const codeHost = new DemoCodeHostAdapter();
  const issueTracker = new DemoIssueTrackerAdapter();

  return { socketName, tmpDir, configPath, statePath, codeHost, issueTracker };
}

export function cleanupDemo(ctx: DemoContext): void {
  try {
    Bun.spawnSync(
      ["tmux", "-L", ctx.socketName, "kill-server"],
      { stdout: "pipe", stderr: "pipe" },
    );
  } catch {}
  try {
    rmSync(ctx.tmpDir, { recursive: true, force: true });
  } catch {}
}
